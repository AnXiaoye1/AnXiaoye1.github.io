利用turtle库绘制一个五角星

要求：

  • 图形的边缘线条为黄色

  • 填充为红色

import turtle

r = 100 # 前进距离
θ1 = 72 # 旋转角度
θ2 = 144
turtle.setup(800,600)
turtle.fillcolor("red") # 填充色为红色
turtle.pensize(5) #画笔宽度
turtle.pencolor("yellow") # 画笔颜色
turtle.begin_fill() #开始填充

# 此时画笔在窗口正中心，需要向左边移动一些距离
turtle.penup()
turtle.fd(-150)
turtle.pendown()

for i in range(5): # 绘图
    turtle.fd(r)
    turtle.left(θ1)
    turtle.fd(r)
    turtle.right(θ2)

turtle.end_fill() #填充结束
turtle.hideturtle() # 隐藏turtle箭头
turtle.done()

利用turtle库绘制数码管显示本地时间

要求：

  • 绘制数码管，显示当前时间（时、分、秒）

  • 格式 18:23:02

# coding:utf-8
# 用turtle绘制七段数码管，来显示电脑本地时间
# 格式如下：
# 15:45:51

import time
import turtle as t

# 绘制间隔
def drawGap():
    t.penup()
    t.fd(5)

# 绘制单段数码管
def drawLine(draw):
    drawGap()
    if(draw):
         t.pendown()
    else:
         t.penup()
    t.fd(50)
    drawGap()
    t.right(90)

# 绘制当前时间
def drawDate(date):
    t.pencolor("red")
    for i in date:
        if i == '=':
            t.write(":", font=("黑体", 50, "normal"))
            t.pencolor("blue")
            t.fd(50)
       elif i == '#':
            t.write(":", font=("黑体", 50, "normal"))
            t.pencolor("green")
            t.fd(50)
       else:
            drawDigit(eval(i))

# 绘制数码管
def drawDigit(cur_time):
    if cur_time in [2, 3, 4, 5, 6, 8, 9]:
        drawLine(True)
    else:
        drawLine(False)

    if cur_time in [0, 1, 3, 4, 5, 6, 7, 8, 9]:
        drawLine(True)
    else:
        drawLine(False)
    if cur_time in [0, 2, 3, 5, 6, 8, 9]:
        drawLine(True)
    else:
        drawLine(False)
    if cur_time in [0, 2, 6, 8]:
        drawLine(True)
    else:
        drawLine(False)
    t.left(90)

    if cur_time in [0, 4, 5, 6, 8, 9]:
        drawLine(True)
    else:
        drawLine(False)
    if cur_time in [0, 2, 3, 5, 6, 7, 8, 9]:
        drawLine(True)
    else:
        drawLine(False)
    if cur_time in [0, 1, 2, 3, 4, 7, 8, 9]:
        drawLine(True)
    else:
        drawLine(False)
    t.left(180)
    t.penup()
    t.fd(20)

# 主函数
def main():
    t.setup(800, 300, 200, 200)
    t.penup()
    t.fd(-275)
    t.pensize(5)
    drawDate(time.strftime('%H=%M#%S', time.localtime()))
    t.hideturtle()
    t.done()

if __name__ == "__main__":
    main()

利用jieba库对《水浒传》进行分析

要求：

  • 分析统计词频排名前十的中文词条
import jieba

txt = open("水浒传.txt", "r", encoding='utf-8').read()
words = jieba.lcut(txt)
counts = {}

for word in words:
    if len(word) == 1:
        continue
    else:
        counts[word] = counts.get(word, 0) + 1
items = list(counts.items())
items.sort(key=lambda x: x[1], reverse=True)

for i in range(10):
    word, count = items[i]
    print("{0:<10}{1:>5}".format(word, count))

求[0,100]内的10个随机数

要求：

  • 计算100内的10个随机数
  • 交换最大值和最小值的位置

# 随机产生[0,100]以内的随机数，找到最大值和最小值并交换位置

import random

a = []
for i in range(10):
    a.append(random.randint(0, 100))
print("随机产生的10个整数：", end="")
print(a)
a_max = max(a)
a_min = min(a)
a[a.index(min(a))] = a_max
a[a.index(max(a))] = a_min
print("最大值：", a_max)
print("最小值：", a_min)
print("交换最大最小值位置：", end="")
print(a)
